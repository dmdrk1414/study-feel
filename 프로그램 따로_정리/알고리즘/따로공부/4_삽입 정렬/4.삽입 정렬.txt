#include <stdio.h>

int main() {
    int temp, j;
    int array[10] = {5, 10, 1, 8, 7, 6, 4, 3, 2, 9};
    
    for(int i = 0; i < 9; i++){
         j = i;
        while(j >= 0 && array[j] > array[ j + 1]){
            temp = array[j];
            array[j] = array[j+1];
            array[j + 1] = temp;
            j--;
        }
    }
    for(int i = 0; i< 10; i++){
        printf("%d", array[i]);
    }
    return 0;
}

#include <stdio.h>

void outPut(int * arr, int len){
    for (int i = 0 ; i < len ; i++){
        printf("[%d]", arr[i]);
    }
    printf("\n");
}

void insert(int *arr ,int n){
    int i, j, key;
    
    for(i=1; i<n ; i++){
        key = arr[i];
        
        for(j=i-1 ; j>=0 && arr[j] > key ; j--){
            arr[j + 1] = arr[j];
        }
        arr[j + 1] = key;
    }
}

int main() {
    int arr[5] = {4,2,3,1,5};
    int len = sizeof(arr)/sizeof(arr[0]);
    outPut(arr,len);
    insert(arr, len - 1);
    printf("\n");
    outPut(arr,len);
    
    return 0;
}

   int number, temp;

        scanf("%d", &arr[i]);
   }
   
   for ( int j = 0 ; j < number ; j ++){
       
       for ( int k = j ; k < number ; k ++){
           if( arr[j] > arr[k]){
               temp = arr[j];
               arr[j] = arr[k];
               arr[k] = temp;
           }
       }
   }
   
   for ( int n = 0 ; n < number ; n ++){

}
