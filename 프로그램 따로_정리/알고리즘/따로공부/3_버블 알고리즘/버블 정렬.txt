#include <stdio.h>

int main(void) {
    int array[10] = {1,10,5,8,7,6,4,3,2,9};
    int temp;
    
    for (int i = 0 ; i < 10 ; i ++){
        for ( int j = 0 ; j < 9-i ; j ++){
            if ( array [j] > array[j + 1]){
                temp = array[j];
                array[j] = array[j + 1];
                array[j + 1] = temp;
            }
        }
    }
    
    for (int k = 0; k < 10 ; k++){
        printf("%d", array[k]);
    }
}


각자 숫자를 뽑아서 0번째와 1번째를 비교를 해서 크면 1번째로 0번째를 복사한다.
