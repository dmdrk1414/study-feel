#include <stdio.h>

int main()
{
    int input;
    scanf_s("%d", &input);

    for (int i = 0; i < input; i++)
    {

        for (int k = 0; k < i; k++)
        {
            printf(" ");
        }

        for (int j = 0; j < (2 * input - 1) - (2 * i); j++)
        {
            printf("*");
        }

        printf("\n");
    }

    return 0;
}
