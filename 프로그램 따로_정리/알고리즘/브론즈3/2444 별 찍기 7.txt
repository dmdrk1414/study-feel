#include <stdio.h>

int main()
{
    int input;
    scanf("%d", &input);

    for (int i = 0; i < input; i++)
    {
        for (int j = 0; j < input - (i + 1); j++)
        {
            printf(" ");
        }
        for (int k = 0; k < (2 * i) + 1; k++)
        {
            printf("*");
        }

        printf("\n");
    }

    for (int n = 0; n < input - 1; n++)
    {
        for (int q = 0; q < n + 1; q++)
        {
            printf(" ");
        }
        for (int w = 0; w < (2 * input - 1) - (2 * (n + 1)); w++)
        {
            printf("*");
        }

        printf("\n");
    }

    return 0;
}
