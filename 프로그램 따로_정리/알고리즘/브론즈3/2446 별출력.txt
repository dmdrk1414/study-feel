#include <stdio.h>

int main()
{

    int input;
    scanf("%d", &input);

    for (int i = 0; i < input; i++)

    {
        for (int k = 1; k < i + 1; k++)
            printf(" ");
        for (int j = 0; j < 2 * input - 1 - 2 * i; j++)
            printf("*");

        printf("\n");
    }

    for (int n = 0; n < 2 * input - 1 - input; n++)
    {

        for (int w = 0; w < input - 2 - n; w++)
        {
            printf(" ");
        }

        for (int p = 0; p < 3 + 2 * n; p++)
        {
            printf("*");
        }

        printf("\n");
    }

    return 0;
}
