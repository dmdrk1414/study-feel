#include <stdio.h>

int main()
{
    int input;
    scanf("%d", &input);

    for (int i = 0; i < input; i++)
    {
        for (int q = 0; q < input - i - 1; q++)
        {
            printf(" ");
        }

        for (int j = 0; j < i + 1; j++)
        {
            printf("*");
        }
        printf("\n");
    }

    return 0;
}
