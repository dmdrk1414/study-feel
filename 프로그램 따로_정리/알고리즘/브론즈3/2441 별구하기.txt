#include <stdio.h>

int main(void)
{
    int input;
    scanf("%d", &input);

    for (int i = 0; i < input; i++)
    {
        for (int k = 0; k < i; k++)
        {
            printf(" ");
        }
        for (int j = 0; j < input - i; j++)
        {
            printf("*");
        }

        printf("\n");
    }
}
