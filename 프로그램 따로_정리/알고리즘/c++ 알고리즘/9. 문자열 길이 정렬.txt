#include<iostream>
#include<algorithm>
using namespace std;

bool compare(string a, string b) {
	if (a.length() < b.length()) { // 내림차순
		return 1;
	} else if (a.length() > b.length()) { // 올림차순은 금지
		return 0;
	} else { // 길이가 같은 경우라면
		return a < b; // 사전순으로 정렬
	}
}

int main() {
	string array[20001];
	int number;
	cin >> number;
	for (int i = 0; i < number; i++) {
		cin >> array[i];
	}

	sort(array, array + number, compare);

	for (int i = 0; i < number; i++) {
		if (i > 0 && array[i] == array[i - 1]) {
			continue;
		} else {
			cout << array[i] << endl;
		}
	}

	return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;
bool cmp(string a, string b)
{
	if (a.size() == b.size())     // 길이가 같으면 사전순으로 반환
		return (a < b);
	else
		return(a.size() < b.size());  내림차순
}

int main()
{
	int N, i;

	cin >> N;
	string arr[N];

	for (i = 0; i < N; i++)
		cin >> arr[i];
	
	sort(arr, arr + N, cmp);
	
	for (i = 0; i < N; i++)
	{
		if (arr[i].compare(arr[i - 1]) != 0)  // compare 함수의 같은게 있으면 0 반환
			cout << arr[i] << '\n';
	}
}
