#include <stdio.h>

int main()
{
    int number;
    long long nValue;
    int result = 0;
    int splitContainer[100] = {
        0,
    };

    while (1)
    {
        scanf_s("%d", &number);
        scanf_s("%lld", &nValue);

        for (int i = 0; nValue > 0; i++)
        {
            splitContainer[i] = nValue % 10;
            nValue = nValue / 10;
        }

        for (int j = 0; j < number; j++)
        {
            result += splitContainer[j];
        }

        for (int k = 0; k < number; k++)
        {
            printf("%d\n", splitContainer[k]);
        }
    }

    /*for (int i = 0; i < number; i++)
    {
        scanf_s("%d", &nValue[i]);
    }

    for (int j = 0; j < number; j++)
    {
        result += nValue[j];
    }*/

    return 0;
}
