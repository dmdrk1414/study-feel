#include <stdio.h>
int main()
{
	int input;
	int add = 1;
	char container[100];

	scanf_s("%d", &input);

	for (int i = 0; i < input; i++)
	{
		int sum = 0;
		int length = 0;

		scanf_s("%s", container, 100);

		for (int j = 0; container[j] != '\0'; j++)
		{
			length += 1;
		}

		for (int k = 0; k < length; k++)
		{
			if (container[k] == 'O')
			{
				sum += add;
				add += 1;
			}
			else
			{
				add = 1;
			}
		}
		add = 1;
		printf("%d\n", sum);
	}

	return 0;
}
