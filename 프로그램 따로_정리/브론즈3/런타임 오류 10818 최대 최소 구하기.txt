#include <stdio.h>

int main()
{
    int input, number[100];
    int max, min;

    scanf("%d", &input);

    for (int i = 0; i < input; i++)
    {
        scanf("%d", &number[i]);
    }

    min = number[0];
    max = number[0];
    for (int j = 0; j < input; j++)
    {
        if (number[j] > max)
        {
            max = number[j];
        }
        if (number[j] < min)
        {
            min = number[j];
        }
    }
    printf("%d %d", min, max);

    return 0;
}
